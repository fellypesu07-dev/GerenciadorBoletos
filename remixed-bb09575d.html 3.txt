const DB = "https://gerenciador-boletos-1d534-default-rtdb.firebaseio.com";

async function fbGet(path) {
  try {
    const r = await fetch(`${DB}/${path}.json`);
    if (!r.ok) throw new Error(r.status);
    return await r.json();
  } catch(e) { setSyncBadge("err","⚠ Erro"); return null; }
}

async function fbPut(path, val) {
  setSyncBadge("saving","Salvando…");
  try {
    const r = await fetch(`${DB}/${path}.json`, {
      method: "PUT",
      headers: {"Content-Type":"application/json"},
      body: JSON.stringify(val)
    });
    if (!r.ok) throw new Error(r.status);
    setSyncBadge("ok","✓ Online");
  } catch(e) { setSyncBadge("err","⚠ Erro"); }
}

async function fbDel(path) {
  setSyncBadge("saving","Salvando…");
  try {
    await fetch(`${DB}/${path}.json`, { method: "DELETE" });
    setSyncBadge("ok","✓ Online");
  } catch(e) { setSyncBadge("err","⚠ Erro"); }
}

// Polling para atualizações em tempo real
let _poll;
async function startPolling() {
  async function load() {
    const [b, c, s, g] = await Promise.all([
      fbGet("boletos"), fbGet("cnpjs"), fbGet("saldos"), fbGet("gcoll")
    ]);
    boletos = b || {};
    cnpjs   = c || {};
    saldos  = s || {};
    gcoll   = g || {};
    setSyncBadge("ok","✓ Online");
    document.getElementById("loadingScreen").style.display = "none";
    document.getElementById("app").style.display = "block";
    renderAll();
  }
  await load();
  _poll = setInterval(load, 8000); // atualiza a cada 8s
}

function setSyncBadge(cls, txt) {
  const el = document.getElementById("syncBadge");
  el.className = "sync-badge sync-" + cls;
  el.textContent = txt;
}

function feriadosAno(y){
  const fx=["01-01","21-04","01-05","07-09","12-10","02-11","15-11","20-11","25-12"].map(s=>`${y}-${s}`);
  function pascoa(yr){const a=yr%19,b=Math.floor(yr/100),c=yr%100,d=Math.floor(b/4),e=b%4,f=Math.floor((b+8)/25),g=Math.floor((b-f+1)/3),h=(19*a+b-d-g+15)%30,i=Math.floor(c/4),k=c%4,l=(32+2*e+2*i-h-k)%7,m=Math.floor((a+11*h+22*l)/451),mes=Math.floor((h+l-7*m+114)/31),dia=(h+l-7*m+114)%31+1;return new Date(yr,mes-1,dia);}
  const p=pascoa(y),add=(d,n)=>{const x=new Date(d);x.setDate(x.getDate()+n);return x;};
  const mv=[add(p,-48),add(p,-47),add(p,-2),p,add(p,60)].map(d=>dstr(d));
